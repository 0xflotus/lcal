    export PATH=/bin:/usr/bin unalias -a

    _print_discordian_year() {

    [ $l -gt 0 ] || l=0

for i in {1..73}
do
  [ $l -eq 5 ] && l=0 && printf "\n"
  if [ $i -lt 10 ]; then
    if [ $i -eq 1 ]; then
      echo -n " ${SEASONS[s++]}  $i"
    elif [ $i -eq 5 ]; then
      echo -n "     *$i"
    else
      echo -n "      $i"
    fi
  elif [ $i -eq 50 ]; then
    echo -n "    *$i"
  else
    echo -n "     $i"
  fi
  ((++l))
done

    }

    _print_gregorian_year() {

    [ $l -gt 0 ] || l=0

for ((i=1; i<=${MONTH_LEN[$1-1]}; i++))
do
  [ $l -eq 5 ] && l=0 && printf "\n"
  if [ $i -lt 10 ]; then
    if [ $i -eq 1 ]; then
      echo -n " ${MONTHS[s++]}  $i"
    else
      echo -n "      $i"
    fi
  else
    echo -n "     $i"
  fi
  ((++l))
done

    }

    _main() {

    if [ $(tput cols) -lt 77 ]; then
  echo The terminal is too small. This chart requires a terminal with
  echo at least 77 columns wide in order to view it properly.
  exit 1
fi

gyear=/tmp/gyear.$$
dyear=/tmp/dyear.$$
convert=gd_chart.txt

SEASONS=(Chs Dsc Cfn Bcy Afm)
MONTHS=(Jan Feb Mar Apr May Jun Jul Aug Sep Oct Nov Dec)
MONTH_LEN=(31 28 31 30 31 30 31 31 30 31 30 31)

l=0 s=0
for n in {1..5}
do
  _print_discordian_year
done > $dyear

l=0 s=0
for n in {1..12}
do
  _print_gregorian_year $n
done > $gyear

>$convert
printf "\n\t\tGregorian\t\t\t\tDiscordian\n\n" >> $convert
printf "\t\t\t\t\t     SM     BT     PD     PP     SO\n" >> $convert
printf "    ------- ------- ------- -------\t" >> $convert
printf "    ------- ------- ------- -------\n" >> $convert

paste $gyear $dyear >> $convert
less $convert
rm $gyear $dyear
rm $convert

    }

    _main
